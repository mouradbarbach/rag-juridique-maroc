ما المقصود بالحكم التحكيمي ؟ يصدر الحكم التحكيمي باغلبيه الاصوات بعد مداوله الهيئه التحكيميه، ويكتسب بمجرد صدوره حجيه الشيء المقضي بخصوص النزاع الذي تم الفصل فيه. يكتسب الحكم التحكيمي بمجرد صدوره حجيه الشيء المقضي به بخصوص النزاع الذي تم الفصل فيه. ولا ينفذ الحكم التحكيمي جبريا الا بمقتضي امر بتخويل الصيغه التنفيذيه يصدره رئيس المحكمه الصادر الحكم في دائرتها. اما الحكم التحكيمي الصادر في الحاله التي يكون فيها احد طرفي النزاع من اشخاص القانون العام، فلا يكتسب حجيه الشيء المقضي به الا بناء علي امر بتخويل الصيغه التنفيذيه. تذييل الاحكام التحكيميه بالصيغه التنفيذيه يختص بامر تخويل الصيغه التنفيذيه رئيس المحكمه الصادر الحكم التحكيمي في دائرتها. واذا تعلق التحكيم باستئناف حكم ما، فان الامر بتخويل الصيغه التنفيذيه يصدره الرئيس الاول لهذه المحكمه. ولا يقبل الامر الصادر بتخويل الصيغه التنفيذيه اي طعن. واذا ما تم رفض تخويل الصيغه التنفيذيه فيجب ان يكون القرار معللا ويمكن استئنافه داخل 15 يوما من تاريخ تبليغه. الطعن في الحكم التحكيمي لا يقبل الحكم التحكيمي اي طعن من طرق الطعن باستثناء اعاده النظر، وتعرض الغير الخارج عن الخصومه. ويمكن ان تكون الاحكام التحكيميه قابله للطعن بالبطلان امام محكمه الاستئناف في حالات محدده، داخل اجل 15 يوما من تبليغ الحكم التحكيمي المذيل بالصيغه التنفيذيه، واذا ابطلت محكمه الاستئناف الحكم التحكيمي تبت في جوهر النزاع في اطار المهمه المسنده الي الهيئه التحكيميه. التحكيم الدولي ما المقصود بالتحكيم الدولي ؟ يعتبر دوليا التحكيم الذي يتعلق بمصالح التجاره الدوليه، والذي يكون لاطرافه علي الاقل موطن او مقر بالخارج. تطبق علي التحكيم الدولي المقتضيات القانونيه المتعلقه بالتحكيم الدولي الوارده بقانون المسطره المدنيه، دون الاخلال بما ورد في الاتفاقيات الدوليه المصادق عليها من لدن المملكه المغربيه والمنشوره بالجريده الرسميه. وزاره العدل والحريات ساحه المامونيه - 10000- الرباط - المغرب الهاتف : 37 37 21 37 05 www.justice.gov.ma / www.mahakim.ma المملكه المغربيه وزاره العدل والعربات الوسائل البديله لحل المنازعات الوسائل البديله لحل المنازعات هي الاليات التي يلجا اليها الاطراف لحل منازعاتهم بواسطه شخص محايد ودون اللجوء الي المحاكم. وتتميز هذه الوسائل بالسرعه وقله التكاليف والسريه، كما تمنح للاطراف مرونه وحريه لا تتوفر عاده في المساطر القضائيه. ويتم تنفيذ هذه الوسائل في اطار مبادئ الاستقلاليه والحياد وشفافيه المساطر والفعاليه. ومن اهم هذه الوسائل، الوساطه الاتفاقيه والتحكيم والصلح. الوساطه الاتفاقيه الصلح التحكيم ماذا نقصد بالتحكيم ؟ يراد بالتحكيم حل نزاع من لدن هيئه تحكيميه تتلقي من الاطراف مهمه الفصل في النزاع بناء علي اتفاق تحكيم بشان الحقوق التي يملكون حريه التصرف فيها. ويمكن ان تكون محل اتفاق تحكيم النزاعات الداخله في اختصاص المحاكم التجاريه، والنزاعات المتعلقه بالعقود التي تبرمها الدوله، والجماعات المحليه، والمقاولات العامه الخاضعه لقانون الشركات التجاريه، والمؤسسات العامه. ولا يمكن ان تكون محل اتفاق تحكيم النزاعات التي تهم حاله الاشخاص واهليتهم، او الحقوق الشخصيه التي لا تكون موضوع تجاره، وكذلك النزاعات المتعلقه بالتصرفات الاحاديه للدوله او الجماعات المحليه او غيرها من الهيئات المتمتعه باختصاصات السلطه العموميه. غير ان النزاعات الماليه الناتجه عن التصرفات الاحاديه المذكوره، يمكن ان تكون محل عقد تحكيم ما عدا المتعلقه بتطبيق قانون جبائي. ماهو اتفاق التحكيم ؟ يقصد باتفاق التحكيم التزام الاطراف باللجوء الي التحكيم قصد حل نزاع نشا، او قد ينشا عن علاقه قانونيه معينه، تعاقديه او غير تعاقديه. يحرر اتفاق التحكيم كتابه، اما بعقد رسمي او عرفي، واما بمحضر يحرر امام الهيئه التحكيميه المختاره. ويعتبر اتفاق التحكيم مبرما كتابه اذا ورد في وثيقه موقعه من الاطراف او في رسائل متبادله، او اتصال بالتلكس، او برقيات، او ايه وسيله اخري من وسائل الاتصال. يكتسي اتفاق التحكيم شكل عقد تحكيم او شرط تحكيم. ما هو عقد التحكيم ؟ هو الاتفاق الذي يلتزم فيه اطراف نزاع نشا بينهم بعرض هذا النزاع علي هيئه تحكيميه. ويمكن ابرامه ولو اثناء دعوي جاريه امام المحكمه. يتضمن عقد التحكيم تحت طائله البطلان : • تحديد موضوع النزاع ؛ • تعيين الهيئه التحكيميه او التنصيص علي طريقه تعيينها. ماهو شرط التحكيم ؟ هو الاتفاق الذي يلتزم فيه اطراف عقد بان يعرضوا علي التحكيم النزاعات التي قد تنشا علي العقد المذكور. ويجب تحت طائله البطلان ان يضمن شرط التحكيم كتابه في العقد الاصلي او في وثيقه تحيل اليه بشكل لا لبس فيه، وان ينص في شرط التحكيم علي تعيين الحكم او علي طريقه تعيينه. ما هي انواع التحكيم ؟ التحكيم اما خاص او مؤسساتي في التحكيم الخاص تتكفل الهيئه التحكيميه بتنظيمه مع تحديد المسطره الواجب اتباعها، ما عدا اذا اتفق الاطراف علي خلاف ذلك او اختاروا نظام تحكيم معين ؛ اما في التحكيم المؤسساتي فتتكفل مؤسسه التحكيم بتنظيم التحكيم وضمان حسن سيره طبقا لنظامها. وتحترم في جميع الاحوال الحقوق المتعلقه بالدفاع، ويلزم المحكمون بكتمان السر المهني. حاله اللجوء الي المحكمه مع وجود اتفاق التحكيم عندما يعرض نزاع مطروح امام هيئه تحكيميه علي نظر احدي المحاكم، يجب عليها التصريح بعدم القبول بشرط ان يدفع المدعي عليه وقبل الدخول في الجوهر بوجود اتفاق تحكيم. واذا كان النزاع لم يعرض بعد علي الهيئه التحكيميه يجب علي المحكمه التصريح بعدم قبول الدعوي متي دفع بذلك المدعي عليه، قبل الدخول في جوهر النزاع وما لم يكن بطلان اتفاق التحكيم واضحا. لا يمنع اتفاق التحكيم اي طرف من اللجوء الي قاضي الامور المستعجله لطلب اتخاذ اي اجراء وقتي او تحفظي. ما هي مهام الهيئه التحكيميه ؟ تختص الهيئه التحكيميه بالبت في صحه وحدود اختصاصها، او في صحه اتفاق التحكيم بامر غير قابل للطعن ؛ تضبط الهيئه التحكيميه اجراءات مسطره التحكيم دون ان تكون ملزمه بتطبيق الاجراءات المتبعه لدي المحاكم، ما لم يتفق الاطراف علي خلاف ذلك في اتفاق التحكيم : تقوم الهيئه التحكيميه بجميع اجراءات التحقيق بالاستماع الي الشهود او بتعيين خبراء او باي اجراء اخر ؛ يحق للهيئه التحكيميه ان تطلب من الطرف الحائز لوسيله اثبات ما الادلاء بها : تتخذ الهيئه التحكيميه ما لم يتم الاتفاق علي خلاف ذلك بطلب من احد الاطراف كل تدبير مؤقت او تحفظي تراه لازما في حدود مهمتها ؛ تطبق هيئه التحكيم في موضوع النزاع القواعد القانونيه التي يتفق عليها الاطراف، وفي حال غياب هذا الاتفاق تطبق هيئه التحكيم القواعد الموضوعيه في القانون الذي تري انه اكثر اتصالا بالنزاع ؛ علي الهيئه التحكيميه ان تراعي في جميع الاحوال شروط العقد موضوع النزاع وتاخذ بعين الاعتبار الاعراف التجاريه والعادات، وما جري عليه التعامل بين الطرفين ؛ واذا اتفق الطرفان علي تفويض هيئه التحكيم صفه وسطاء بالتراضي، تفصل الهيئه في هذه الحاله في موضوع النزاع بناء علي قواعد العداله والانصاف دون التقيد بالقانون.
